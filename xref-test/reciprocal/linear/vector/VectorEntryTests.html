9   final class VectorEntryTests {
10      @Test
11      void constructor_should_throw_Exception_when_index_is_0() {
12          assertThatIllegalArgumentException().isThrownBy(() -> new VectorEntry<>(0, 0))
13              .withMessage("index > 0 expected but index = 0").withNoCause();
14      }
15  
16      @Test
17      void constructor_should_throw_Exception_when_element_is_null() {
18          assertThatNullPointerException().isThrownBy(() -> new VectorEntry<Integer>(1, null)).withMessage("element")
19              .withNoCause();
20      }
21  
22      @Test
23      void withIndex_should_throw_Exception_when_newIndex_is_0() {
24          final var entry = new VectorEntry<>(1, 0);
25  
26          assertThatIllegalArgumentException().isThrownBy(() -> entry.withIndex(0))
27              .withMessage("index > 0 expected but index = 0").withNoCause();
28      }
29  
30      @Test
31      void withIndex_should_return_copy() {
32          final var entry = new VectorEntry<>(1, 0);
33  
34          final var actual = entry.withIndex(2);
35  
36          assertThat(actual.index()).isEqualByComparingTo(2);
37          assertThat(actual.element()).isZero();
38      }
39  
40      @Test
41      void withElement_should_throw_Exception_when_newElement_is_null() {
42          final var entry = new VectorEntry<>(1, 0);
43  
44          assertThatNullPointerException().isThrownBy(() -> entry.withElement(null)).withMessage("newElement")
45              .withNoCause();
46      }
47  
48      @Test
49      void withElement_should_return_copy() {
50          final var entry = new VectorEntry<>(1, 0);
51  
52          final var actual = entry.withElement(1);
53  
54          assertThat(actual.index()).isOne();
55          assertThat(actual.element()).isOne();
56      }
