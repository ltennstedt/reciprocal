8   final class PolarFormTests {
9       @Test
10      void equalsByComparing_should_throw_Exception_when_other_is_other() {
11          final var polarForm = new PolarForm(0.0D, 0.0D);
12  
13          assertThatNullPointerException().isThrownBy(() -> polarForm.equalsByComparing(null)).withMessage("other")
14              .withNoCause();
15      }
16  
17      @Test
18      void equalsByComparing_should_return_false_when_radials_are_unequal_by_comparing() {
19          final var polarForm = new PolarForm(0.0D, 0.0D);
20          final var other = new PolarForm(1.0D, 0.0D);
21  
22          assertThat(polarForm.equalsByComparing(other)).isFalse();
23      }
24  
25      @Test
26      void equalsByComparing_should_return_false_when_angulars_are_unequal_by_comparing() {
27          final var polarForm = new PolarForm(0.0D, 0.0D);
28          final var other = new PolarForm(0.00D, 1.0D);
29  
30          assertThat(polarForm.equalsByComparing(other)).isFalse();
31      }
32  
33      @Test
34      void equalsByComparing_should_return_true_when_this_equals_by_comparing_to_other() {
35          final var polarForm = new PolarForm(0.0D, 0.0D);
36          final var other = new PolarForm(0.00D, 0.00D);
37  
38          assertThat(polarForm.equalsByComparing(other)).isTrue();
39      }
