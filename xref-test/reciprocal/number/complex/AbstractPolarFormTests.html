6   import java.math.BigDecimal;

10  final class AbstractPolarFormTests {
11      @Test
12      void constructor_should_throw_Exception_when_radial_is_null() {
13          assertThatNullPointerException().isThrownBy(() -> new BigPolarForm(null, null)).withMessage("radial")
14              .withNoCause();
15      }
16  
17      @Test
18      void constructor_should_throw_Exception_when_angular_is_null() {
19          assertThatNullPointerException().isThrownBy(() -> new BigPolarForm(BigDecimal.ZERO, null))
20              .withMessage("angular").withNoCause();
21      }
22  
23      @Test
24      void doesNotEqualByComparing_should_throw_Exception_when_other_is_null() {
25          assertThatNullPointerException()
26              .isThrownBy(() -> new BigPolarForm(BigDecimal.ZERO, BigDecimal.ZERO).doesNotEqualByComparing(null))
27              .withMessage("other").withNoCause();
28      }
29  
30      @Test
31      void doesNotEqualByComparing_should_false_when_they_equals_by_comparing() {
32          final var polarForm = new BigPolarForm(BigDecimal.ZERO, BigDecimal.ZERO);
33          final var other = new BigPolarForm(BigDecimal.ZERO.setScale(1, RoundingMode.HALF_UP), BigDecimal.ZERO);
34  
35          assertThat(polarForm.doesNotEqualByComparing(other)).isFalse();
36      }
37  
38      @Test
39      void doesNotEqualByComparing_should_true_when_they_do_not_equal_by_comparing() {
40          final var polarForm = new BigPolarForm(BigDecimal.ZERO, BigDecimal.ZERO);
41          final var other = new BigPolarForm(BigDecimal.ONE, BigDecimal.ZERO);
42  
43          assertThat(polarForm.doesNotEqualByComparing(other)).isTrue();
44      }
