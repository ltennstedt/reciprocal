3   import static com.google.common.base.Preconditions.checkArgument;
4   import static java.util.Objects.requireNonNull;
5   
6   import java.io.Serial;
7   import java.io.Serializable;
8   import java.math.BigDecimal;
9   import java.util.Comparator;
10  import org.jetbrains.annotations.NotNull;
11  import reciprocal.ReciprocalUtils;
12  
13  /**
14   * Immutable implementation of a circle which uses {@link BigDecimal} as type
15   * for its radius
16   *
17   * @since 0.0.1
18   */
19  public final class BigCircle extends AbstractCircle<BigDecimal, BigCircle> {
20      @Serial
21      private static final long serialVersionUID = 1L;
22  
23      /**
24       * Constructor
25       *
26       * @param radius radius
27       * @throws NullPointerException when {@code radius == null}
28       * @throws IllegalArgumentException when {@code radius <= 0}
29       * @since 0.0.1
30       */
31      public BigCircle(final @NotNull BigDecimal radius) {
32          super(radius);
33          checkArgument(radius.compareTo(BigDecimal.ZERO) > 0, "radius > 0 expected but radius = %s", radius);
34      }
35  
36      @Override
37      public @NotNull BigDecimal getDiameter() {
38          return BigDecimal.valueOf(2L).multiply(getRadius());
39      }
40  
41      @Override
42      public @NotNull BigDecimal getCircumference() {
43          return ReciprocalUtils.BIG_PI.multiply(getDiameter());
44      }
45  
46      @Override
47      public @NotNull BigDecimal getArea() {
48          return ReciprocalUtils.BIG_PI.multiply(getRadius().pow(2));
49      }
50  
51      @Override
52      public int compareTo(final @NotNull BigCircle o) {
53          requireNonNull(o, "o");
54          return BigCircleComparator.INSTANCE.compare(this, o);
55      }
56  
57      /**
58       * Comparator
59       *
60       * @since 0.0.1
61       */
62      public static final class BigCircleComparator implements Comparator<@NotNull BigCircle>, Serializable {
63          /**
64           * Instance
65           *
66           * @since 0.0.1
67           */
68          public static final @NotNull BigCircleComparator INSTANCE = new BigCircleComparator();
69  
70          @Serial
71          private static final long serialVersionUID = 1L;
72  
73          private BigCircleComparator() {
74          }
75  
76          @Override
77          public int compare(final @NotNull BigCircle o1, final @NotNull BigCircle o2) {
78              requireNonNull(o1, "o1");
79              requireNonNull(o2, "o2");
80              return o1.getRadius().compareTo(o2.getRadius());
81          }
82      }
83  }
