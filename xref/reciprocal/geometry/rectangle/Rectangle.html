1   package reciprocal.geometry.rectangle;

3   import static com.google.common.base.Preconditions.checkArgument;
4   import static java.util.Objects.requireNonNull;
5   
6   import java.io.Serial;
7   import org.jetbrains.annotations.NotNull;
8   
9   /**
10   * Immutable implementation of a rectangle which uses {@link Double} as type for
11   * its length and width
12   *
13   * @since 0.0.1
14   */
15  public final class Rectangle extends AbstractRectangle<Double, Rectangle> {
16      @Serial
17      private static final long serialVersionUID = 1L;
18  
19      /**
20       * Constructor
21       *
22       * @param length length
23       * @param width width
24       * @throws IllegalArgumentException when {@code length <= 0}
25       * @throws IllegalArgumentException when {@code width <= 0}
26       * @since 0.0.1
27       */
28      public Rectangle(final double length, final double width) {
29          super(length, width);
30          checkArgument(length > 0.0D, "length > 0 expected but length = %s", length);
31          checkArgument(width > 0.0D, "width > 0 expected but width = %s", width);
32      }
33  
34      @Override
35      public boolean isSquare() {
36          return getLength().compareTo(getWidth()) == 0;
37      }
38  
39      @Override
40      public @NotNull Double getPerimeter() {
41          return 2.0D * (getLength() + getWidth());
42      }
43  
44      @Override
45      public @NotNull Double getDiagonal() {
46          return Math.sqrt(Math.pow(getLength(), 2.0D) + Math.pow(getWidth(), 2.0D));
47      }
48  
49      @Override
50      public @NotNull Rectangle withLength(final @NotNull Double newLength) {
51          requireNonNull(newLength, "newLength");
52          checkArgument(newLength > 0.0D, "newLength > 0 expected but newLength = %s", newLength);
53          return new Rectangle(newLength, getWidth());
54      }
55  
56      @Override
57      public @NotNull Rectangle withWidth(final @NotNull Double newWidth) {
58          requireNonNull(newWidth, "newWidth");
59          checkArgument(newWidth > 0.0D, "newWidth > 0 expected but newWidth = %s", newWidth);
60          return new Rectangle(getLength(), newWidth);
61      }
62  }
