4   import static java.util.Objects.requireNonNull;

6   import java.io.Serial;
7   import java.util.List;

10  /**
11   * Immutable implementation of a vector which uses {@link Long} as type for its elements
12   *
13   * @since 0.0.1
14   */
15  public final class LongVector extends AbstractVector<@NotNull Long, @NotNull LongVector, @NotNull Double> {
16      @Serial
17      private static final long serialVersionUID = 1L;
18  
19      LongVector(final int size, final @NotNull List<@NotNull VectorEntry<@NotNull Long>> entries) {
20          super(size, entries);
21      }
22  
23      /**
24       * Returns a {@link LongVectorBuilder}
25       *
26       * @param size size
27       * @return {@link LongVectorBuilder}
28       * @throws IllegalArgumentException when {@code size < 1}
29       * @since 0.0.1
30       */
31      public static LongVectorBuilder ofSize(final int size) {
32          return new LongVectorBuilder(size);
33      }
34  
35      @Override
36      public @NotNull LongVector add(final @NotNull LongVector summand) {
37          requireNonNull(summand, "summand");
38          checkArgument(getSize() == summand.getSize(), "equal sizes expected but %s != %s", getSize(),
39              summand.getSize());
40          return new LongVector(getSize(),
41              getEntries().map(e -> e.withElement(e.element() + summand.getElement(e.index()))).toList());
42      }
43  
44      @Override
45      public @NotNull LongVector subtract(final @NotNull LongVector subtrahend) {
46          requireNonNull(subtrahend, "subtrahend");
47          checkArgument(getSize() == subtrahend.getSize(), "equal sizes expected but %s != %s", getSize(),
48              subtrahend.getSize());
49          return new LongVector(getSize(),
50              getEntries().map(e -> e.withElement(e.element() - subtrahend.getElement(e.index()))).toList());
51      }
52  
53      @Override
54      public @NotNull Long dotProduct(final @NotNull LongVector other) {
55          requireNonNull(other, "other");
56          checkArgument(getSize() == other.getSize(), "equal sizes expected but %s != %s", getSize(), other.getSize());
57          return getEntries().map(e -> e.element() * other.getElement(e.index())).reduce(Long::sum)
58              .orElseThrow();
59      }
60  
61      @Override
62      public @NotNull LongVector scalarMultiply(final @NotNull Long scalar) {
63          requireNonNull(scalar, "scalar");
64          return new LongVector(getSize(), getEntries().map(e -> e.withElement(scalar * e.element())).toList());
65      }
66  
67      @Override
68      public @NotNull LongVector negate() {
69          return new LongVector(getSize(), getEntries().map(e -> e.withElement(-e.element())).toList());
70      }
71  
72      @Override
73      public boolean orthogonalTo(final @NotNull LongVector other) {
74          requireNonNull(other, "other");
75          checkArgument(getSize() == other.getSize(), "equal sizes expected but %s != %s", getSize(), other.getSize());
76          return dotProduct(other) == 0L;
77      }
78  
79      @Override
80      public @NotNull Double taxicabNorm() {
81          return getElements().map(Math::abs).reduce(Long::sum).map(Long::doubleValue).orElseThrow();
82      }
83  
84      @Override
85      public @NotNull Double euclideanNorm() {
86          return Math.sqrt(euclideanNormPow2());
87      }
88  
89      @Override
90      public @NotNull Double maxNorm() {
91          return getElements().map(Math::abs).max(Long::compareTo).map(Long::doubleValue).orElseThrow();
92      }
93  
94      @Override
95      protected @NotNull Double euclideanNormPow2() {
96          return getElements().map(e -> e * e).reduce(Long::sum).map(Long::doubleValue).orElseThrow();
97      }
98  
99      /**
100      * Builder for {@link LongVector LongVectors}
101      *
102      * @since 0.0.1
103      */
104     public static final class LongVectorBuilder extends AbstractVectorBuilder<Long, LongVector, LongVectorBuilder> {
105         LongVectorBuilder(final int size) {
106             super(size, i -> 0L);
107         }
108 
109         @Override
110         public @NotNull LongVector build() {
111             return new LongVector(getSize(), computeEntries());
112         }
113     }
114 }
